go slice

2021年12月20日
| go内部源码浅析

slice是区别于固定长度的数组而有的一种可自动扩容的、内置的基础的数据类型。 slice在使用的时候通常遇到的问题就是下标越界，其他的如果不清除他底层的原理的话可能会带给你一些意想不到的bug，比方说数据被修改，更大的问题的话就是内存泄漏的问题了。

package main

import (
	"fmt"
	"math/rand"
	"time"
)

var cards = []int32{1, 2, 3, 4, 5}

func GetCards() []int32 {
	var myCards []int32
	myCards = cards

	for i := 0; i < len(cards); i++ {
		exchangePos := rand.Intn(len(cards))
		if i != exchangePos {
			myCards[i], myCards[exchangePos] = myCards[exchangePos], myCards[i]
		}
	}
	return myCards
}

func main() {
	rand.Seed(time.Now().UnixNano())
	fmt.Printf("cards:		%v\n", cards)
	fmt.Printf("-----------------------\n")
	fmt.Printf("my cards:	%v\n", GetCards())
	fmt.Printf("cards:		%v\n", cards)
	/*
		output:
			cards:		[1 2 3 4 5]
			-----------------------
			my cards:	[1 4 2 5 3]
			cards:		[1 4 2 5 3]
	*/
}

package main

import (
	"fmt"
	"math/rand"
	"time"
)

var cards = []int32{1, 2, 3, 4, 5}

func GetCards() []int32 {
	var myCards []int32
	myCards = cards[1:]
	return myCards
}

func main() {
	rand.Seed(time.Now().UnixNano())
	fmt.Printf("cards:	%v\n", cards)
	fmt.Printf("-----------------------\n")
	card2 := GetCards()
	fmt.Printf("card2:	%v\n", card2)
	card2[0] = 8
	fmt.Printf("-----------------------\n")
	fmt.Printf("cards:	%v\n", cards)
	fmt.Printf("cards2:	%v\n", card2)
	fmt.Printf("cards:%p	cards2:%p\n", &cards[1], &card2[0])
	/*
		output:
			cards:	[1 2 3 4 5]
			-----------------------
			card2:	[2 3 4 5]
			-----------------------
			cards:	[1 8 3 4 5]
			cards2:	[8 3 4 5]
			cards:0x100e884	cards2:0x100e884
	*/
}

上面两个例子都是修改了原始的slice，第一个例子当然很明显，如果业务代码里面隐藏的类似第二种情况的话一开始可能会有些抓脑，这个就是问题的现象，不论是单一的slice对象还是复杂对象里面的slice都有同样的问题。

内存泄漏

内存泄漏对于不需要手动管理内存的语言来说基本是用这舒服，大部分情况下面都不用care这个，即使遇到了估计也要程序到了一定量或时间的积累才会把问题暴露出来，重启一下可好？

内存管理是一个非常大的问题，操作系统也有自己的一套管理内存的方法，这种带GC的也是有自己的一套管理方法，为什么要有呢，就是解放程序员的负担。让你值关系你的业务就好了，大体思路就是预先分配一块内存，然后根据自己的一套方法再来管理这些内存的使用，在GC的时候在扫描所有对象把空闲的内存进行释放掉。后续有心情可以了解下内存分配和调度系统的实现^^，了解一下也可以解解乏,^^！！

由于内存是go自己先向系统申请，然后自己来管理内存的分配和释放，我们在编写代码的时候在对内存的使用的时候尽可能不要大内存的对象被占用小内存的对象持有并得不到释放。

一下是模拟的一块内存，slice类型的a对象一开始拥有这块内存，后面b通过[pos:]指向了同一块内存，后面
---------------------------------------------------------------------------------------------------------------------------------------------
	|	|	|	|	|	|	|	|	|	|	|	|	|	|	|	|	|	
---------------------------------------------------------------------------------------------------------------------------------------------
a:
													b:

由于b也指向了之前a内存地址的某一段，所有就会导致有空于的一段不能好的利用，长此以往就会越来越多，所以我们也可以用append(a[pos:],1,3,2,3,4)避免这个问题。

func main() {
	sliceA := make([]int, 0, 8)
	sliceA = append(sliceA, 1, 2, 3, 4, 5, 6, 7, 8)
	sliceB := sliceA[4:]
	sliceC := append(sliceA[4:], 9, 10)
	fmt.Printf("cap(sliceB) %v\n", cap(sliceB))
	hb := (*reflect.SliceHeader)(unsafe.Pointer(&sliceB))
	ha := (*reflect.SliceHeader)(unsafe.Pointer(&sliceA))
	hc := (*reflect.SliceHeader)(unsafe.Pointer(&sliceC))
	fmt.Printf("%+v \n%+v\n%+v\n", *ha, *hb, *hc)
}
// output:
// cap(sliceB) 4
// {Data:824634564288 Len:8 Cap:8} 
// {Data:824634564320 Len:4 Cap:4}
// {Data:824634523904 Len:6 Cap:8}

type slice struct {
	array unsafe.Pointer
	len   int
	cap   int
}

func growslice(et *_type, old slice, cap int) slice {
	//.....

	if cap < old.cap {
		panic(errorString("growslice: cap out of range"))
	}

	if et.size == 0 {
		// append should not create a slice with nil pointer but non-zero len.
		// We assume that append doesn't need to preserve old.array in this case.
		return slice{unsafe.Pointer(&zerobase), old.len, cap}
	}

	newcap := old.cap
	doublecap := newcap + newcap
	if cap > doublecap {
		newcap = cap
	} else {
		if old.cap < 1024 {
			newcap = doublecap
		} else {
			// Check 0 < newcap to detect overflow
			// and prevent an infinite loop.
			for 0 < newcap && newcap < cap {
				newcap += newcap / 4
			}
			// Set newcap to the requested cap when
			// the newcap calculation overflowed.
			if newcap <= 0 {
				newcap = cap
			}
		}
	}

	var overflow bool
	var lenmem, newlenmem, capmem uintptr
	// Specialize for common values of et.size.
	// For 1 we don't need any division/multiplication.
	// For sys.PtrSize, compiler will optimize division/multiplication into a shift by a constant.
	// For powers of 2, use a variable shift.
	switch {
	case et.size == 1:
		lenmem = uintptr(old.len)
		newlenmem = uintptr(cap)
		capmem = roundupsize(uintptr(newcap))
		overflow = uintptr(newcap) > maxAlloc
		newcap = int(capmem)
	case et.size == sys.PtrSize:
		lenmem = uintptr(old.len) * sys.PtrSize
		newlenmem = uintptr(cap) * sys.PtrSize
		capmem = roundupsize(uintptr(newcap) * sys.PtrSize)
		overflow = uintptr(newcap) > maxAlloc/sys.PtrSize
		newcap = int(capmem / sys.PtrSize)
	case isPowerOfTwo(et.size):
		var shift uintptr
		if sys.PtrSize == 8 {
			// Mask shift for better code generation.
			shift = uintptr(sys.Ctz64(uint64(et.size))) & 63
		} else {
			shift = uintptr(sys.Ctz32(uint32(et.size))) & 31
		}
		lenmem = uintptr(old.len) << shift
		newlenmem = uintptr(cap) << shift
		capmem = roundupsize(uintptr(newcap) << shift)
		overflow = uintptr(newcap) > (maxAlloc >> shift)
		newcap = int(capmem >> shift)
	default:
		lenmem = uintptr(old.len) * et.size
		newlenmem = uintptr(cap) * et.size
		capmem, overflow = math.MulUintptr(et.size, uintptr(newcap))
		capmem = roundupsize(capmem)
		newcap = int(capmem / et.size)
	}

	// The check of overflow in addition to capmem > maxAlloc is needed
	// to prevent an overflow which can be used to trigger a segfault
	// on 32bit architectures with this example program:
	//
	// type T [1<<27 + 1]int64
	//
	// var d T
	// var s []T
	//
	// func main() {
	//   s = append(s, d, d, d, d)
	//   print(len(s), "\n")
	// }
	if overflow || capmem > maxAlloc {
		panic(errorString("growslice: cap out of range"))
	}

	var p unsafe.Pointer
	if et.ptrdata == 0 {
		p = mallocgc(capmem, nil, false)
		// The append() that calls growslice is going to overwrite from old.len to cap (which will be the new length).
		// Only clear the part that will not be overwritten.
		memclrNoHeapPointers(add(p, newlenmem), capmem-newlenmem)
	} else {
		// Note: can't use rawmem (which avoids zeroing of memory), because then GC can scan uninitialized memory.
		p = mallocgc(capmem, et, true)
		if lenmem > 0 && writeBarrier.enabled {
			// Only shade the pointers in old.array since we know the destination slice p
			// only contains nil pointers because it has been cleared during alloc.
			bulkBarrierPreWriteSrcOnly(uintptr(p), uintptr(old.array), lenmem-et.size+et.ptrdata)
		}
	}
	memmove(p, old.array, lenmem)

	return slice{p, old.len, newcap}
}

func main() {
	sliceA := make([]int, 0, 1)

看源码的结构就只有三个成员，存放元素的内存地址array，元素的个数len，以及这个slice的最大容量cap。结合我们的例子，内存的重新分配是在slice容量不够的时候才会去发生的。所有他的原理也比较简单，算出一个新的容量，在分配一块新的内存，然后通过memmove把原有地址上面的东西复制过去，相信写过C的你会很有感觉。

注意事项

  1. 在使用的时候尽量用append去操作slice
  2. 在编写业务代码的时候结合map和slice成员用指针来使用
  3. 有并发读写的情况下面要使用锁或者其他方式来解决
