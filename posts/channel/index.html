go channel

2021年12月16日
| go内部源码浅析
| 阅读

channel通常理解和把他当成程序内部的队列来使用，根据队列的长度，可以把队列分为有缓冲队列(长度非0),也叫不阻塞队列；无缓冲队列(长度为0)，也叫阻塞队列。

我们在项目实践过程种最多的用到的还是在不同goroutine之间传递数据，可以是一对一或者多对一又或者多对多，这取决于自己的业务场景里面解决的什么问题。

package main

import (
	"fmt"
	"time"
)

type Bus struct {
	Id      int64
	Code    string
	Company string
}

type People struct {
	Name string
}

// 限定只读的写法
// func WaitBus(wait <-chan Bus) {
func WaitBus(wait chan Bus, people People) {
	fmt.Printf("%v is waiting...\n", people.Name)
	bus := <-wait
	fmt.Printf("bus come %+v\n", bus)
}

func main() {
	var (
		busChannel chan Bus
		bus        Bus = Bus{Id: 1, Code: "01A543", Company: "江湖再见"}
	)
	busChannel = make(chan Bus) // 不指定后面的size就是阻塞的
	go WaitBus(busChannel, People{Name: "p1"})
	busChannel <- bus
	time.Sleep(2 * time.Second) //
	fmt.Printf("welcome!!")
}

package main

import (
	"fmt"
	"time"
)

type Bus struct {
	Id      int64
	Code    string
	Company string
}

type People struct {
	Name string
}

// 限定只读的写法
// func WaitBus(wait <-chan Bus) {
func WaitBus(wait chan Bus, people People) {
	for bus := range wait { // 另外一种写法
		fmt.Printf("%v get in %+v\n", people.Name, bus)
	}
}

func main() {
	var (
		busChannel chan Bus
	)
	busChannel = make(chan Bus, 4) // 4辆巴士
	for i := 0; i < 10; i++ {      // 10个人等bus
		go WaitBus(busChannel, People{Name: fmt.Sprintf("p%v", i+1)})
	}

	for i := 0; i < 4; i++ {
		go func(i int) {
			busChannel <- Bus{Id: int64(i + 1), Company: "江湖再见"}
		}(i)
	}
	time.Sleep(2 * time.Second) //
	/*
		output:
			p7 get in {Id:3 Code: Company:江湖再见}
			p1 get in {Id:4 Code: Company:江湖再见}
			p2 get in {Id:2 Code: Company:江湖再见}
			p5 get in {Id:1 Code: Company:江湖再见}
	*/
}

type hchan struct {
	qcount   uint           // total data in the queue
	dataqsiz uint           // size of the circular queue
	buf      unsafe.Pointer // points to an array of dataqsiz elements
	elemsize uint16
	closed   uint32
	elemtype *_type // element type
	sendx    uint   // send index
	recvx    uint   // receive index
	recvq    waitq  // list of recv waiters
	sendq    waitq  // list of send waiters

	// lock protects all fields in hchan, as well as several
	// fields in sudogs blocked on this channel.
	//
	// Do not change another G's status while holding this lock
	// (in particular, do not ready a G), as this can deadlock
	// with stack shrinking.
	lock mutex
}

我们看到有一个 buf成员指针，指向一个数组，数组的长度为dataqsize，数组存放的内容的类型是elemtype,还有当前有多少个数据在队列里面，以及往队列里面发送数据和等待读取队列的goroutine的两个双向链表，看起来差不多了懂了的样子，还有两个看起来不是很明确用来做什么的变量sendx recvx,接下来我们看下看往队列里面放数据和读取数据是怎么实现的。

func chansend(c *hchan, ep unsafe.Pointer, block bool, callerpc uintptr) bool {
	// ..... 
	if c.qcount < c.dataqsiz {
		// Space is available in the channel buffer. Enqueue the element to send.
		qp := chanbuf(c, c.sendx)
		if raceenabled {
			racenotify(c, c.sendx, nil)
		}
		typedmemmove(c.elemtype, qp, ep)
		c.sendx++
		if c.sendx == c.dataqsiz {
			c.sendx = 0
		}
		c.qcount++
		unlock(&c.lock)
		return true
	}
}

func chanrecv(c *hchan, ep unsafe.Pointer, block bool) (selected, received bool) {
    // ....
    	if sg := c.sendq.dequeue(); sg != nil {
		// Found a waiting sender. If buffer is size 0, receive value
		// directly from sender. Otherwise, receive from head of queue
		// and add sender's value to the tail of the queue (both map to
		// the same buffer slot because the queue is full).
		recv(c, sg, ep, func() { unlock(&c.lock) }, 3)
		return true, true
	}

	if c.qcount > 0 {
		// Receive directly from queue
		qp := chanbuf(c, c.recvx)
		if raceenabled {
			racenotify(c, c.recvx, nil)
		}
		if ep != nil {
			typedmemmove(c.elemtype, ep, qp)
		}
		typedmemclr(c.elemtype, qp)
		c.recvx++
		if c.recvx == c.dataqsiz {
			c.recvx = 0
		}
		c.qcount--
		unlock(&c.lock)
		return true, true
	}
}

type waitq struct {
	first *sudog
	last  *sudog
}

func (q *waitq) enqueue(sgp *sudog) {
	sgp.next = nil
	x := q.last
	if x == nil {
		sgp.prev = nil
		q.first = sgp
		q.last = sgp
		return
	}
	sgp.prev = x
	x.next = sgp
	q.last = sgp
}

func (q *waitq) dequeue() *sudog {
	for {
		sgp := q.first
		if sgp == nil {
			return nil
		}
		y := sgp.next
		if y == nil {
			q.first = nil
			q.last = nil
		} else {
			y.prev = nil
			q.first = y
			sgp.next = nil // mark as removed (see dequeueSudog)
		}

		// if a goroutine was put on this queue because of a
		// select, there is a small window between the goroutine
		// being woken up by a different case and it grabbing the
		// channel locks. Once it has the lock
		// it removes itself from the queue, so we won't see it after that.
		// We use a flag in the G struct to tell us when someone
		// else has won the race to signal this goroutine but the goroutine
		// hasn't removed itself from the queue yet.
		if sgp.isSelect && !atomic.Cas(&sgp.g.selectDone, 0, 1) {
			continue
		}

		return sgp
	}
}

我们看过发送和接收这两个函数后，队列里面有数据的情况下面主要是队当前队列长度以及发送和接收的下标移动来实现数据的收发逻辑，核心逻辑大体也是这样的，数组长度是固定的，只要在这个数组上面通过下标的移动来实现这个channel的功能。其他的话主体也是发送的时候如果满了就放到sendq，接收如果队列目前没数据的话放到recvq

						|sendx 为6，当前队列qcount为6个
	|recvx 为1，当前队列qcount为5

这样就实现了一个队列的功能了，主题功能的实现看似简单，但是结合goroutine的调度可能就没那么容易了，需要结合当前goroutine已经channel当前的一个特性进行不同逻辑的实现，总体而言，只要有了方向和合理的实现方案，我想那些代码在我们的“疑问"中最终都会得到答案的！

注意事项

  1. 不要往已经关闭的channel里面放东西
  2. 不要 重复关闭
  3. 在往channel写数据的时候不论是阻塞的还是非阻塞的最好用select来避免阻塞当前goroutine
